class Solution:
    def minWindow(self, s: str, t: str) -> str:
        length, res, left, right = len(s), s, 0, 0

        def isHas(x: str, y: str) -> bool:
            x, y = Counter(x), Counter(y)
            for item in y:
                if y[item] > x[item]:
                    return False
            return True

        if not isHas(s, t):
            return ''
        else:
            while right <= length:
                si = s[left:right]
                print('si: ', si)
                if isHas(si, t):
                    while isHas(si, t):
                        left += 1
                        si = s[left:right]
                        print('     left loop: ', si)
                    res = res if len(res) < len(s[left - 1:right]) else s[left - 1:right]
                    print("         res: ", res)
                right += 1
        return res

                right += 1
